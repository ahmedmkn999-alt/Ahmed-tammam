rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // ✅ الطلاب: الطالب يقدر يقرأ بيانات نفسه بس
    match /students/{studentId} {
      allow create: if request.auth == null &&
                       request.resource.data.keys().hasAll(['fullName', 'studentPhone', 'password']) &&
                       request.resource.data.password is string &&
                       request.resource.data.password.size() == 64; // SHA-256 hex
      // ❌ مفيش قراءة أو تعديل من الـ client بدون Firebase Auth
      allow read, update, delete: if false;
    }

    // ✅ أكواد التفعيل: قراءة للكود المحدد فقط (للفحص)، كتابة ممنوعة من الـ client
    match /activation_codes/{codeId} {
      allow read: if true; // لازم يقرأ عشان يفحص الكود
      allow update: if request.resource.data.diff(resource.data).affectedKeys()
                       .hasOnly(['used', 'usedBy', 'usedAt']); // بس الحقول دي
      allow create, delete: if false; // الـ admin فقط عن طريق Firebase Admin SDK
    }

    // ✅ رسائل الدعم: أي حد يكتب، مفيش حد يقرأ إلا الـ admin
    match /support_messages/{msgId} {
      allow create: if request.resource.data.keys().hasAll(['subject', 'body']) &&
                       request.resource.data.body.size() < 5000;
      allow read, update, delete: if false;
    }

    // ❌ كل باقي الـ collections ممنوعة
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
